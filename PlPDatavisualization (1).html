def load_and_prepare_data():
    """
    Load the Iris dataset, convert it to a DataFrame,
    and perform basic cleaning and inspection.
    """
    try:
        # Load Iris dataset from sklearn
        iris = load_iris()
        df = pd.DataFrame(iris.data, columns=iris.feature_names)
        df['species'] = pd.Categorical.from_codes(iris.target, iris.target_names)

        print("\n📌 First five rows of the dataset:")
        print(df.head())

        print("\n📌 Dataset Info:")
        print(df.info())

        # Check for missing values
        print("\n🔍 Missing values in each column:")
        print(df.isnull().sum())

        # Cleaning: Drop rows with missing values (if any)
        df.dropna(inplace=True)

        return df

    except Exception as e:
        print(f"❌ Error loading dataset: {e}")
        return None

def analyze_data(df):
    """
    Perform basic statistical analysis on the dataset.
    """
    print("\n📊 Descriptive Statistics:")
    print(df.describe())

    print("\n📊 Mean of features grouped by species:")
    group_mean = df.groupby("species").mean()
    print(group_mean)

def visualize_data(df):
    """
    Visualize the dataset using various plots.
    """
    sns.set(style="whitegrid")

    # Line Plot: Sepal Length trend across index (simulated time)
    plt.figure(figsize=(10, 5))
    df["index"] = df.index
    sns.lineplot(x="index", y="sepal length (cm)", hue="species", data=df)
    plt.title("Sepal Length Trend Across Index (Simulated Time)")
    plt.xlabel("Index")
    plt.ylabel("Sepal Length (cm)")
    plt.legend(title="Species")
    plt.tight_layout()
    plt.show()

    # Bar Plot: Average petal length per species
    plt.figure(figsize=(7, 5))
    sns.barplot(x="species", y="petal length (cm)", data=df, estimator="mean")
    plt.title("Average Petal Length by Species")
    plt.ylabel("Average Petal Length (cm)")
    plt.tight_layout()
    plt.show()

    # Histogram: Distribution of sepal width
    plt.figure(figsize=(7, 5))
    sns.histplot(df["sepal width (cm)"], kde=True, color="green", bins=20)
    plt.title("Distribution of Sepal Width")
    plt.xlabel("Sepal Width (cm)")
    plt.tight_layout()
    plt.show()

    # Scatter Plot: Sepal Length vs. Petal Length
    plt.figure(figsize=(8, 6))
    sns.scatterplot(x="sepal length (cm)", y="petal length (cm)", hue="species", data=df)
    plt.title("Sepal Length vs. Petal Length")
    plt.xlabel("Sepal Length (cm)")
    plt.ylabel("Petal Length (cm)")
    plt.legend(title="Species")
    plt.tight_layout()
    plt.show()

def main():

    Main function to run the data loading, analysis, and visualization.
    """
    df = load_and_prepare_data()
    if df is not None:
        analyze_data(df)
        visualize_data(df)
